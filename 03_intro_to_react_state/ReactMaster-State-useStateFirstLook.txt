import { useState } from "react";
import "./styles.css";

function Header() {
  return <header className="App"></header>;
}

function MainContainer() {
  const [selectedItem, setSelectedItem] = useState(null);

  function onSelectedItem(item) {
    setSelectedItem(item);
  }

  return (
    <>
      <Basket selectedItem={selectedItem} />
      <ProductsContainer onSelectedItem={onSelectedItem} />
    </>
  );
}

function Basket({ selectedItem }) {
  return (
    <div>
      <h4>
        {selectedItem
          ? `You selected ${selectedItem.name} with id ${selectedItem.id}`
          : "Basket is empty"}
      </h4>
    </div>
  );
}

function ProductsContainer({ onSelectedItem }) {
  const items = [
    { id: 1, name: "Apples" },
    { id: 2, name: "Bananas" },
    { id: 3, name: "Cherries" },
  ];

  return <Products items={items} onClick={onSelectedItem} />;
}

function Products({ items, onClick }) {
  return (
    <ul>
      {items.map((item) => (
        <li key={item.id} onClick={() => onClick(item)}>
          {item.name}
        </li>
      ))}
    </ul>
  );
}

function LogoContainer({ children }) {
  return (
    <>
      <div>Copyright My Company</div>
      {children}
    </>
  );
}

function Logo({ dimensions }) {
  return (
    <>
      <img
        src={CircleSvg}
        alt="My lovely circle"
        width={dimensions.width}
        height={dimensions.height}
      />
    </>
  );
}

function Footer() {
  return (
    <footer>
      <div>This is the footer</div>
    </footer>
  );
}

const App = () => {
  return (
    <>
      <Header />
      <MainContainer />
      <Footer />
    </>
  );
};

export default App;
